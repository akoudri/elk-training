# Create admin role
POST /_security/role/admin
{
  "cluster": ["all"],
  "indices": [
    {
      "names": ["*"],
      "privileges": ["all"]
    }
  ]
}

# Create User
POST /_security/user/john
{
  "password" : "password123",
  "roles" : [ "admin" ],
  "full_name" : "John Doe"
}

# Role Mapping
POST /_security/role_mapping/admin
{
  "roles": [ "admin" ],
  "users": [ "john" ]
}
